import Image from 'next/image'
import { useEffect, useRef } from 'react'

export function WatercolourSplash() {
  return (
    <svg
      aria-hidden="true"
      style={{ position: 'absolute', top: '-10px', right: '-10px', pointerEvents: 'none', opacity: 0.18, zIndex: 0 }}
      width="260" height="140" viewBox="0 0 260 140"
    >
      <ellipse cx="160" cy="70" rx="140" ry="60" fill="#ff7a85" filter="url(#watercolor-splash)" />
      <ellipse cx="60" cy="90" rx="70" ry="40" fill="#ffb3b8" filter="url(#watercolor-splash)" />
    </svg>
  )
}

export function Gallery() {
  const ref = useRef(null)

  useEffect(() => {
    const el = ref.current
    if (!el) return

    let isDown = false
    let startX, scrollLeft
    let velocity = 0
    let lastX = 0
    let rafId = null

    const cancelMomentum = () => {
      if (rafId) cancelAnimationFrame(rafId)
    }

    const applyMomentum = () => {
      if (Math.abs(velocity) < 0.5) return
      el.scrollLeft += velocity
      velocity *= 0.92
      rafId = requestAnimationFrame(applyMomentum)
    }

    const onMouseDown = (e) => {
      isDown = true
      cancelMomentum()
      startX = e.pageX - el.offsetLeft
      scrollLeft = el.scrollLeft
      lastX = e.pageX
      velocity = 0
      el.style.scrollSnapType = 'none'
    }

    const onMouseUp = () => {
      if (!isDown) return
      isDown = false
      el.style.scrollSnapType = ''
      rafId = requestAnimationFrame(applyMomentum)
    }

    const onMouseLeave = () => {
      if (!isDown) return
      isDown = false
      el.style.scrollSnapType = ''
      rafId = requestAnimationFrame(applyMomentum)
    }

    const onMouseMove = (e) => {
      if (!isDown) return
      e.preventDefault()
      const x = e.pageX - el.offsetLeft
      velocity = lastX - e.pageX
      lastX = e.pageX
      el.scrollLeft = scrollLeft + (startX - x)
    }

    el.addEventListener('mousedown', onMouseDown)
    el.addEventListener('mouseleave', onMouseLeave)
    el.addEventListener('mouseup', onMouseUp)
    el.addEventListener('mousemove', onMouseMove)
    return () => {
      cancelMomentum()
      el.removeEventListener('mousedown', onMouseDown)
      el.removeEventListener('mouseleave', onMouseLeave)
      el.removeEventListener('mouseup', onMouseUp)
      el.removeEventListener('mousemove', onMouseMove)
    }
  }, [])

  const paintings = [
    { src: '/images/beach.jpg', alt: 'Painting of a beach scene', width: 800, height: 600 },
    { src: '/images/rock.jpg', alt: 'Painting of rocks along the coast', width: 600, height: 600 },
    { src: '/images/boat.jpg', alt: 'Painting of a boat on the water', width: 800, height: 600 },
    { src: '/images/spain.webp', alt: 'Painting inspired by Spanish architecture and streets', width: 600, height: 800 },
    { src: '/images/house.jpg', alt: 'Painting of a house', width: 800, height: 600 },
    { src: '/images/jug.jpg', alt: 'Painting of a ceramic jug', width: 600, height: 800 },
  ]

  return (
    <div className="art-gallery" ref={ref}>
      {paintings.map(({ src, alt, width, height }) => (
        <Image
          key={src}
          src={src}
          alt={alt}
          width={width}
          height={height}
          sizes="600px"
          draggable={false}
          style={{ height: '380px', width: 'auto', maxWidth: 'none', objectFit: 'cover', borderRadius: '4px', margin: 0, flexShrink: 0 }}
        />
      ))}
    </div>
  )
}
